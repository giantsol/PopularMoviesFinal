2. PreferenceUtils
    private UNKNOWN_LAST_ORDERING = -1
    private KEY_LAST_ORDERING = "last_ordering"

    int getLastOrdering()
        int lastOrdering = getPreferences().getInt(KEY_LAST_ORDERING, UNKNOWN_LAST_ORDERING)
        if (lastOrdering == UNKNOWN_LAST_ORDERING) lastOrdering = DEFAULT_ORDERING
        return lastOrdering
